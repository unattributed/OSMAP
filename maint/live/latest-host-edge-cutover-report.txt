########################################################################
# 0. Suricata-assisted IPS guardrails (tables defined in pf.conf)
########################################################################

# Tables are fed by suricata_eve2pf.ksh (watch first, block later).
# suricata_watch / suricata_block / suricata_allow are defined globally in pf.conf.

########################################################################
# 2. WAN facing rules on the external interface (IPv4 and IPv6)
########################################################################

# 2.1 SSH from anywhere on the internet (rate limited, fed into <sshguard>)
pass in quick on $ext_if inet proto tcp from any to ($ext_if) port $ssh_port \
    flags S/SA \
    keep state (max-src-conn-rate 10/60, overload <sshguard> flush global, \
                source-track rule, if-bound, src.track 60)

# 2.2 WireGuard listener on the public internet
pass in quick on $ext_if inet proto udp from any to ($ext_if) port $wg_port \
    keep state (if-bound)

# SMTP ingress is handled in the spamd anchor (port 25 policy lives there)

# 2.3 Public HTTPS for OSMAP at the canonical mail host root

    flags S/SA synproxy state (if-bound)

# 2.4 Explicitly drop the remaining end-user mail ports on the WAN
# keep them reachable only via the reviewed browser edge or WireGuard
block drop in log quick on $ext_if inet proto tcp from any \
    to ($ext_if) port $wan_blocked_tcp_svcs

########################################################################
# 3. Services for WireGuard clients (10.44.0.0/24, IPv4 only)
########################################################################

# - to the wg0 address

    port $vpn_host_tcp_svcs \
    flags S/SA synproxy state (if-bound)

pass in quick on $vpn_if inet proto tcp from $vpn_net to ($ext_if) \
    port $vpn_host_tcp_svcs \
    flags S/SA keep state (if-bound)

    keep state (if-bound)
